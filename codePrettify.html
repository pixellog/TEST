	기본 : pre.prettyprint
	라인 표시할 때 : pre.prettyprint.linenums
	페이지 전체 코드 표시할 때 : pre.prettyprint.linenums.html

	(function ($) {
		var prettify = (function () {
			function html(s) {
				return s.replace(/&/g, '&').replace(//g, '>');
			}

			var el = document.getElementsByTagName('pre');
			$.each(el, function (i) {
				if (el[i].className !== 'prettyprint linenums html') {
					var quineHtml = html(el[i].innerHTML);
				} else {
					var quineHtml = html('\n\n' + document.documentElement.innerHTML + '\n<\/html>\n');
				}
				return el[i].innerHTML = quineHtml;
			});
		}());
	})(jQuery);
